function divide(num, val integer, switch bool) real ans; {
    if (switch == true) {
        ans = num / val;
    } else {
        ans = val / num;
    }
    endif

    ret ans;
}

function multiply(num, val integer, switch bool) integer ans; {
    if (switch == true) {
        ans = num * val;
    } else ans = -3;
    endif

    ret ans;
}

#
integer num, val;
bool switch;
real ans;

get(num, val, switch);

ans = divide(num, val, switch);

put(ans);

ans = multiply(num, val, switch);

if (ans == -3) {
    ret 0;
} else {
    put(ans);
}
endif
#
